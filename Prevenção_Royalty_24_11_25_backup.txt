                        // Faz scroll para topo APÓS mudar de página, com múltiplos delays
                        setTimeout(scrollParaTopoGlobal, 100);
                        setTimeout(scrollParaTopoGlobal, 300);
                        setTimeout(scrollParaTopoGlobal, 600);
                    }
                }, 3000); // 3 segundos para ver o prêmio
            } else {

                setTimeout(() => {

        function validarPagina(numero) {
            switch(numero) {
                case 1:
                    const allConsentimentosGerais = document.getElementById('consentimento1').checked &&
                           document.getElementById('consentimento2').checked &&
                           document.getElementById('consentimento3').checked &&
                           document.getElementById('consentimento4').checked &&
                           document.getElementById('consentimento5').checked &&
                           document.getElementById('consentimento6').checked &&
                           document.getElementById('consentimento7').checked &&
                           document.getElementById('consentimento8').checked;
                    if (!allConsentimentosGerais) {
                        mostrarErroValidacao('Por favor, confirme TODOS os consentimentos gerais antes de continuar.');
                    }
                    return allConsentimentosGerais;
                
                case 1.5:
                    const allConsentimentosSaude = document.getElementById('consentimentoSaude1').checked &&
                           document.getElementById('consentimentoSaude2').checked &&
                           document.getElementById('consentimentoSaude3').checked &&
                           document.getElementById('consentimentoSaude4').checked &&
                           document.getElementById('consentimentoSaude5').checked &&
                           document.getElementById('consentimentoSaude6').checked;
                    if (!allConsentimentosSaude) {
                        mostrarErroValidacao('Por favor, confirme TODOS os consentimentos de saúde mental antes de continuar. Sua privacidade é essencial!');
                    }
                    return allConsentimentosSaude;
                
                case 2:
                    const nome = document.getElementById('nomeCompleto').value.trim();
                    const email = document.getElementById('email').value.trim();
                    const telefone = document.getElementById('telefone').value.trim();
                    const diagnostico = document.getElementById('diagnostico').value.trim();
                    const dataAlta = document.getElementById('dataAlta').value;
                    const terapeuta = document.getElementById('terapeuta').value.trim();

                    if (!nome) {
                        mostrarErroValidacao('❌ Nome completo é obrigatório.');
                        return false;
                    }
                    if (!email) {
                        mostrarErroValidacao('❌ Email é obrigatório.');
                        return false;
                    }
                    if (!validarEmail(email)) {
                        mostrarErroValidacao('❌ Email inválido. Use formato: seu@email.com');
                        return false;
                    }
                    if (!telefone) {
                        mostrarErroValidacao('❌ Telefone é obrigatório.');
                        return false;
                    }
                    if (!diagnostico) {
                        mostrarErroValidacao('❌ Diagnóstico/Condição é obrigatório.');
                        return false;
                    }
                    if (!dataAlta) {
                        mostrarErroValidacao('❌ Data da alta é obrigatória.');
                        return false;
                    }
                    if (!terapeuta) {
                        mostrarErroValidacao('❌ Nome do terapeuta é obrigatório.');
                        return false;
                    }
                    return true;
                
                default:
                    return true;
            }
        }

        function abrirPDFEmJanela(pdfBlob, nomeArquivo) {
            try {
                // Validar que é um Blob válido
                if (!(pdfBlob instanceof Blob)) {
                    throw new Error('PDF blob inválido');
                }
                
                mostrarCarregamento('📄 Abrindo PDF...');
                
                // Criar URL para o PDF
                const pdfUrl = URL.createObjectURL(pdfBlob);
                
                // Abrir em nova aba para visualização
                const novaJanela = window.open(pdfUrl, '_blank');
                
                // Se a janela foi bloqueada pelo popup blocker, tentar download como fallback
                if (!novaJanela || novaJanela.closed || typeof novaJanela.closed === 'undefined') {
                    console.warn('⚠️ Janela bloqueada. Tentando download como fallback...');
                    
                    // Fallback: fazer download se pop-up foi bloqueado
                    const pdfLink = document.createElement('a');
                    pdfLink.href = pdfUrl;
                    pdfLink.download = nomeArquivo;
                    document.body.appendChild(pdfLink);
                    pdfLink.click();
                    document.body.removeChild(pdfLink);
                    
                    fecharCarregamento();
                    alert('⚠️ Pop-up foi bloqueado.\n\nArquivo baixado para sua pasta Downloads:\n' + nomeArquivo);
                    
                } else {
                    // Se conseguiu abrir, mostrar instruções
                    setTimeout(() => {
                        fecharCarregamento();
                        alert('📄 PDF Aberto!\n\nNa janela do navegador:\n• Use o botão "Imprimir" (Ctrl+P / Cmd+P) para:\n  ✓ Imprimir na sua impressora\n  ✓ Salvar como PDF no dispositivo\n\n💡 Você pode salvar em qualquer local do seu dispositivo.');
                        
                        // Liberar URL após um tempo
                        setTimeout(() => {
                            URL.revokeObjectURL(pdfUrl);
                            console.log('📥 PDF visualizado pelo usuário:', nomeArquivo);
                        }, 30000); // Manter URL por 30 segundos
                        
                    }, 800);
                }
                
            } catch (error) {
                fecharCarregamento();
                console.error('❌ Erro ao abrir PDF:', error);
                alert('❌ Erro ao abrir PDF: ' + error.message + '\n\nPor favor, tente novamente.');
            }
        }

        function calcularIdade() {
            const dataNascimento = document.getElementById('dataNascimento').value;
            const idadeInput = document.getElementById('idade');

            if (!dataNascimento) {
                idadeInput.value = '';
                return;
            }

            const data = new Date(dataNascimento);
            const hoje = new Date();
            let idade = hoje.getFullYear() - data.getFullYear();
            const mes = hoje.getMonth() - data.getMonth();

            if (mes < 0 || (mes === 0 && hoje.getDate() < data.getDate())) {
                idade--;
            }

            idadeInput.value = idade >= 0 ? idade : '';
        }

        function mostrarCarregamento(mensagem = 'Processando...') {
            let overlay = document.getElementById('loadingOverlay');
            if (!overlay) {
                overlay = document.createElement('div');
                overlay.id = 'loadingOverlay';
                overlay.style.cssText = 'position: fixed; top: 0; left: 0; right: 0; bottom: 0; background: rgba(0,0,0,0.6); display: flex; align-items: center; justify-content: center; z-index: 3000;';
                overlay.innerHTML = '<div style="background: white; padding: 30px 50px; border-radius: 12px; text-align: center; box-shadow: 0 8px 32px rgba(0,0,0,0.3);"><div style="font-size: 24px; margin-bottom: 15px;">⏳</div><div id="loadingMessage" style="font-size: 16px; color: #333; font-weight: 600;">Processando...</div></div>';
                document.body.appendChild(overlay);
            }
            document.getElementById('loadingMessage').textContent = mensagem;
        }

        function fecharCarregamento() {
            const overlay = document.getElementById('loadingOverlay');
            if (overlay) {
                overlay.remove();
            }
        }

        // BAIXAR ARQUIVOS LOCALMENTE
        // ============================================

        function baixarArquivos() {
            const dados = coletarTodosDados();
            const nomeArquivo = `Relatorio_Prevencao_${dados.nomeCompleto.replace(/\s+/g, '_')}_${new Date().toISOString().slice(0, 10)}`;

            mostrarCarregamento('📄 Preparando relatório PDF...');

            // Usar setTimeout para não bloquear a thread
            setTimeout(() => {
                try {
                    // Gerar PDF
                    const pdfHtml = gerarPDF();
                    const element = document.createElement('div');
                    element.innerHTML = pdfHtml;
                    // Largura compatível com A4 (210mm ≈ 800px a 96dpi)
                    element.style.cssText = 'width: 100%; max-width: 800px; margin: 0 auto; padding: 0; box-sizing: border-box; overflow: visible;';
                    
                    const opt = {
                        margin: 10,
                        filename: `${nomeArquivo}.pdf`,
                        image: { type: 'jpeg', quality: 0.95 },
                        html2canvas: { 
                            scale: 2,
                            useCORS: true,
                            allowTaint: true,
                            windowWidth: 800,
                            windowHeight: element.scrollHeight,
                            backgroundColor: '#ffffff'
                        },
                        jsPDF: { 
                            orientation: 'portrait', 
                            unit: 'mm', 
                            format: 'a4',
                            compress: true
                        }
                    };

                    html2pdf().set(opt).from(element).output('blob').then((pdfBlob) => {
                        setTimeout(() => {
                            // ✅ Abrir PDF em janela para visualização/salvamento
                            
                            try {
                                if (!(pdfBlob instanceof Blob)) {
                                    throw new Error('PDF Blob inválido');
                                }
                                
                                abrirPDFEmJanela(pdfBlob, `${nomeArquivo}.pdf`);
                                fecharCarregamento();
                                console.log('✅ PDF aberto para visualização:', nomeArquivo);
                            } catch (blobError) {
                                fecharCarregamento();
                                console.error('❌ Erro ao processar PDF:', blobError);
                                alert('❌ Erro ao processar PDF: ' + blobError.message);
                            }
                        }, 300);
                    }).catch(error => {
                        fecharCarregamento();
                        console.error('❌ Erro ao gerar PDF:', error);
                        alert('⚠️ Erro ao gerar PDF: ' + error.message);
                    });
                } catch (error) {
                    fecharCarregamento();
                    alert('⚠️ Erro: ' + error.message);
                }
            }, 100);
        }
